import matplotlib.pyplot as plt
numbers = [12, 27, 3, 19, 34, 8, 21, 5, 30, 16,
           2, 25, 11, 35, 7, 18, 29, 14, 9, 22,
           4, 31, 17, 6, 28, 13, 24, 1, 20, 33]
# X values (index positions)
x = list(range(1, len(numbers) + 1))
plt.scatter(x, numbers)
plt.title("Scatter Plot of Given Numbers")
plt.xlabel("Index Position")
plt.ylabel("Values")
plt.grid()
plt.show()
